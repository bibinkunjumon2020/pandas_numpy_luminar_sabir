#identity matrix - DIAGONAL ELEMENTS MUST BE 1 & No.of rows and columns same ex:3*3,4*4

import numpy as np
matrix=np.eye(3,dtype=int)
print(matrix)

matrix=np.eye(3,3,dtype=int)  #eye
print(matrix)

matrix=np.identity(2,dtype=int)   #identity
print(matrix)
